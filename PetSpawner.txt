local function showRedirectLoadingScreen(callback)
    -- Create loading screen for redirection
    local loadingGui = Instance.new("ScreenGui")
    loadingGui.Name = "RedirectLoading"
    loadingGui.Parent = LocalPlayer.PlayerGui
    loadingGui.ResetOnSpawn = false

    local frame = Instance.new("Frame", loadingGui)
    frame.Size = UDim2.new(0, 260, 0, 120)
    frame.Position = UDim2.new(0.5, -130, 0.5, -60)
    frame.BackgroundColor3 = Color3.fromRGB(34, 8, 48)
    frame.BorderSizePixel = 0
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 18)
    local stroke = Instance.new("UIStroke", frame)
    stroke.Color = Color3.fromRGB(150, 0, 255)
    stroke.Thickness = 2
    stroke.Transparency = 0.17

    local label = Instance.new("TextLabel", frame)
    label.Size = UDim2.new(1, 0, 0, 40)
    label.Position = UDim2.new(0, 0, 0, 18)
    label.BackgroundTransparency = 1
    label.Text = "Loading..."
    label.Font = Enum.Font.FredokaOne
    label.TextSize = 20
    label.TextColor3 = Color3.fromRGB(220, 120, 255)

    local progressBarBack = Instance.new("Frame", frame)
    progressBarBack.Size = UDim2.new(0.8, 0, 0, 22)
    progressBarBack.Position = UDim2.new(0.1, 0, 0, 70)
    progressBarBack.BackgroundColor3 = Color3.fromRGB(50, 30, 80)
    progressBarBack.BorderSizePixel = 0
    Instance.new("UICorner", progressBarBack).CornerRadius = UDim.new(0, 12)

    local progressBar = Instance.new("Frame", progressBarBack)
    progressBar.Size = UDim2.new(0, 0, 1, 0)
    progressBar.BackgroundColor3 = Color3.fromRGB(128, 56, 255)
    progressBar.BorderSizePixel = 0
    Instance.new("UICorner", progressBar).CornerRadius = UDim.new(0, 12)

    for i = 1, 50 do
        progressBar.Size = UDim2.new(i/50, 0, 1, 0)
        wait(0.08)
    end

    wait(0.6)
    loadingGui:Destroy()
    callback()
end

-- Minimize toggle
local minimized = false
minimizeBtn.MouseButton1Click:Connect(function()
    minimized = not minimized
    if minimized then
        TweenService:Create(content, TweenInfo.new(0.2), {Size = UDim2.new(1, 0, 0, 0)}):Play()
        TweenService:Create(main, TweenInfo.new(0.2), {Size = UDim2.new(0, 340, 0, 42)}):Play()
    else
        TweenService:Create(content, TweenInfo.new(0.2), {Size = UDim2.new(1, 0, 1, -46)}):Play()
        TweenService:Create(main, TweenInfo.new(0.2), {Size = UDim2.new(0, 340, 0, 270)}):Play()
    end
end)

-- Hotkey toggle (Right Shift)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if not gameProcessed and input.KeyCode == Enum.KeyCode.RightShift then
        gui.Enabled = not gui.Enabled
    end
end)
